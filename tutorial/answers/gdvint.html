% gdvint

GEMPAK-GDVINT>l                   
 GDFILE   = $HDS/96030312_eta.gem
 GDOUTF   = eta_theta.grd
 GDATTIM  = f000
 GVCORD   = pres/thta
 GLEVEL   = 280-350-5             
 MAXGRD   = 2000
 GAREA    = @1;1;93;65
 VCOORD   = pres;none;trop;maxw;sgma
